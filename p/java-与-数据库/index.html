public int executeUpdate(String sql, Object... params) throws SQLException{
    int result = 0;
    stat = conn.prepareStatement(sql);
    for(int i=0;i<prepareStatement.length;i++){
        stat.setObject(i+1;Params[i]);
    }
    result = stat.executeUpdate();
    return result;
}
