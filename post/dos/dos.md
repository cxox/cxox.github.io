# cmd

```bash
# 盘符切换
F:

# 目录
dir

# 切换目录
cd
cd /d F: # /d 可以实现跨盘的操作
cd .. # 返回上一级

# 清理屏幕
cls # clear screen

# 退出
exit

# 查看电脑IP
ipconfig

# 打开软件，直接用软件名
notepad # 记事本
calc # 计算器
mspaint # 画图

# ping命令,用来测试网络是否正常
ping www.xoxzzz.com
# 创建test文件夹
md test

# 创建test.txt文件
cd>test.txt

# 删除文件
del test.txt
rm test.txt

# 删除目录
rd test
rm -r test.txt

# ...

```
